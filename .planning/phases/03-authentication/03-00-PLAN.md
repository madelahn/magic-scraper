Create `jest.config.ts` at project root:

```typescript
import type { Config } from 'jest'
import { pathsToModuleNameMapper } from 'ts-jest'

const config: Config = {
  preset: 'ts-jest',
  testEnvironment: 'node',
  roots: ['<rootDir>/tests'],
  setupFiles: ['<rootDir>/jest.setup.ts'],
  moduleNameMapper: pathsToModuleNameMapper(
    { '@/*': ['./src/*'] },
    { prefix: '<rootDir>/' }
  ),
}

export default config
```

Key details:
- `testEnvironment: 'node'` because all auth tests run server-side (no DOM needed)
- `roots: ['<rootDir>/tests']` keeps test files outside src/
- `moduleNameMapper` maps `@/*` to `./src/*` matching the project's tsconfig paths
- `setupFiles` loads jest.setup.ts before each test file
